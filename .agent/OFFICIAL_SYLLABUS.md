### **1. Module**
```javascript
{
  _id: ObjectId,
  name: String, // "English & Communication Skills"
  icon: String,
  description: String,
  order: Number,
  isActive: Boolean
}
```

### **2. Category**
```javascript
{
  _id: ObjectId,
  module: ObjectId, // ref: Module
  name: String, // "Verbal Ability"
  description: String,
  order: Number
}
```

### **3. Topic**
```javascript
{
  _id: ObjectId,
  category: ObjectId, // ref: Category
  name: String, // "Vocabulary"
  shortNotes: String, // Concise markdown notes
  keyPoints: [String],
  formulas: [String],
  tips: [String],
  difficulty: String,
  order: Number,
  totalQuestions: Number
}
```

### **4. Question**
```javascript
{
  _id: ObjectId,
  topic: ObjectId,
  category: ObjectId,
  module: ObjectId,
  questionType: String, // "MCQ", "Coding", "Descriptive"
  question: String,
  options: [{
    label: String,
    text: String
  }],
  correctAnswer: String,
  explanation: String,
  quickTip: String,
  difficulty: String, // "Easy", "Medium", "Hard"
  timeToSolve: Number,
  tags: [String],
  companies: [String],
  yearAsked: Number,
  similarQuestions: [ObjectId]
}
```

### **5. TestPattern**
```javascript
{
  _id: ObjectId,
  name: String, // "AMCAT Pattern"
  company: String,
  description: String,
  sections: [{
    module: ObjectId,
    category: ObjectId,
    name: String,
    questionCount: Number,
    timeLimit: Number,
    difficulty: String
  }],
  totalQuestions: Number,
  totalTime: Number,
  cutoff: Number,
  isActive: Boolean
}
```

### **6. TestAttempt**
```javascript
{
  _id: ObjectId,
  user: ObjectId,
  testPattern: ObjectId,
  startedAt: Date,
  completedAt: Date,
  status: String, // "in-progress", "completed", "abandoned"
  sections: [{
    module: ObjectId,
    category: ObjectId,
    questions: [{
      question: ObjectId,
      selectedAnswer: String,
      isCorrect: Boolean,
      timeTaken: Number,
      isMarked: Boolean,
      isSkipped: Boolean
    }],
    score: Number,
    accuracy: Number,
    timeSpent: Number
  }],
  totalScore: Number,
  percentage: Number,
  percentile: Number,
  rank: Number,
  strengths: [String],
  weaknesses: [String]
}
```
